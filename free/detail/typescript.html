Let's Make a Color Changer!

Click the buttons to change the circle's color

// This TypeScript code makes the color change magic happen!
function changeColor(color: string) {
  const circle = document.getElementById('colorCircle');
  if (circle) {
    circle.style.backgroundColor = color;
  }
}
