How Do You Make a Chart?

magic-chart-maker.js

// First, get your numbers ready

let iceCreamVotes = [15, 23, 8, 12];

let flavors = ["Chocolate", "Vanilla", "Strawberry", "Mint"];

 

// Then, ask Chart.js to make a pretty picture

new Chart(ctx, {

  type: 'bar',

  data: {

    labels: flavors,

    datasets: [{ data: iceCreamVotes }]

  }

});

 

// Ta-da! You made a chart! 🎉
