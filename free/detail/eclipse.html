Bouncing Ball

This code makes a ball bounce around the screen!

let ball = {
  x: 50,
  y: 50,
  size: 20,
  speedX: 3,
  speedY: 2
};

function moveBall() {
  // Move the ball
  ball.x += ball.speedX;
  ball.y += ball.speedY;
  
  // Bounce off walls
  if (ball.x > width || ball.x < 0) {
    ball.speedX = -ball.speedX;
  }
  if (ball.y > height || ball.y < 0) {
    ball.speedY = -ball.speedY;
  }
}

Colorful Shapes

This code draws different shapes with different colors!

function drawShapes() {
  // Draw a square
  fill("red");
  square(20, 20, 50);
  
  // Draw a circle
  fill("blue");
  circle(120, 45, 50);
  
  // Draw a triangle
  fill("yellow");
  triangle(200, 70, 230, 20, 260, 70);
}
