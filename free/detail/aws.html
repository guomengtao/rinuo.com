simple-aws-example.js 

// Let's create something cool with AWS!
function buildCloudCastle() {
  let castle = new AWS.CloudCastle();
  castle.addTower("Web Tower");
  castle.addTower("Game Tower");
  castle.addMoat("Data River");
  
  return castle;
}

// Wow, look at our castle!
let myCastle = buildCloudCastle();
myCastle.makeItRainConfetti();
