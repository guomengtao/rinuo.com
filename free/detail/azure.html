Fun with Code

Let's try some simple code that Azure can help run!

hello-azure.js
// This is a simple Azure program
function sayHello(name) {
    return `Hello, ${name}! Welcome to Azure!`;
}

// Let's try it with your name
let myName = "Friend";
let message = sayHello(myName);

// See what happens!
console.log(message);
